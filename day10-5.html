Make a bar plot of the months in which movies with "Christmas" In their title tend to be released In the USA.¶

In [15]:
import pandas as pd
import matplotlib.pyplot as plt

christmas_movies = release_dates[release_dates['title'].str.contains('Christmas', case=False)]
usa_christmas_movies = christmas_movies[christmas_movies['country'] == 'USA']
usa_christmas_movies['month'] = pd.to_datetime(usa_christmas_movies['year'], format='%Y').dt.month_name()
monthly_counts = usa_christmas_movies['month'].value_counts().sort_index()

plt.bar(monthly_counts.index, monthly_counts.values)
plt.xlabel('Month')
plt.ylabel('Number of Movies')
plt.title('Movies with "Christmas" in Title Released in USA by Month')
plt.xticks(rotation=45)
plt.show()

judi_dench_movies = release_dates[release_dates['title'].str.contains('Judi Dench', case=False)]

# Convert the 'release_date' column to a datetime data type
judi_dench_movies['date'] = pd.to_datetime(judi_dench_movies['date'])

# Extract the month from the 'release_date' and count occurrences
month_counts = judi_dench_movies['date'].dt.month_name().value_counts()

# Sort the months by their chronological order
month_order = [
    'January', 'February', 'March', 'April', 'May', 'June',

]
month_counts = month_counts.reindex(month_order)

# Print the results
print("Months in which films with Judi Dench tend to be released in the USA:")
print(month_counts)
